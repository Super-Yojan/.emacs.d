(defvar my-typst-watch-process nil)

(defun my-typst-compile-and-view ()
  "Compile current Typst file and open PDF."
  (interactive)
  (when buffer-file-name
    (save-buffer)
    (call-process "typst" nil nil nil "compile" buffer-file-name)
    (let ((pdf (concat (file-name-sans-extension buffer-file-name) ".pdf")))
      (when (file-exists-p pdf)
        (call-process (if (eq system-type 'darwin) "open" "xdg-open") nil nil nil pdf)))))

(defun my-typst-watch-toggle ()
  "Toggle typst watch mode."
  (interactive)
  (if (and my-typst-watch-process (process-live-p my-typst-watch-process))
      (progn (kill-process my-typst-watch-process)
             (setq my-typst-watch-process nil)
             (message "Typst watch stopped"))
    (when buffer-file-name
      (save-buffer)
      (setq my-typst-watch-process
            (start-process "typst-watch" "*typst-watch*" "typst" "watch" "--open" buffer-file-name))
      (message "Typst watch started"))))

(defun my-typst-view-pdf ()
  "View PDF for current Typst file."
  (interactive)
  (when buffer-file-name
    (let ((pdf (concat (file-name-sans-extension buffer-file-name) ".pdf")))
      (if (file-exists-p pdf)
          (call-process (if (eq system-type 'darwin) "open" "xdg-open") nil nil nil pdf)
        (message "PDF not found. Compile first with SPC m c")))))

;; ── texfrag (LaTeX preview in any buffer) ──
(use-package texfrag
  :commands (texfrag-mode texfrag-document texfrag-clear-document)
  :config (define-key texfrag-mode-map (kbd "C-c C-x C-l") 'texfrag-document))

(defvar-local my-texfrag-previews-active nil)

(defun my-latex-preview-toggle ()
  "Toggle LaTeX preview in current buffer."
  (interactive)
  (cond
   ((derived-mode-p 'org-mode) (call-interactively 'org-latex-preview))
   ((bound-and-true-p texfrag-mode)
    (if my-texfrag-previews-active
        (progn (texfrag-clear-document) (setq my-texfrag-previews-active nil) (message "Previews cleared"))
      (progn (texfrag-document) (setq my-texfrag-previews-active t) (message "Previews rendered"))))
   (t (message "Enable texfrag-mode first (M-x texfrag-mode)"))))

(global-set-key (kbd "C-c l p") 'my-latex-preview-toggle)
